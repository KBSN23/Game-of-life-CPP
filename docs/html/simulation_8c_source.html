256void nextTick(nextTickArgs args)
257{
258 simState *simulation = args.simulation;
259 int **cellsMatrix = args.cellsMatrix;
260
261 if (simulation->isPaused)
262 {
263 return; // Jeśli symulacja jest zatrzymana, opuść funkcję
264 }
265
266 int **tempMatrix = cellsMatrix_create(); // Utwórz tymczasową macierz
267
268 // Skopiuj zawartość macierzy komórek do macierzy tymczasowej
269 for (int i = 0; i < BOARD_ROWS; i++)
270 {
271 for (int j = 0; j < BOARD_COLS; j++)
272 {
273 tempMatrix[i][j] = cellsMatrix[i][j];
274 }
275 }
276
277 // Przejdź przez każdą komórkę w macierzy
278 for (int i = 0; i < BOARD_ROWS; i++)
279 {
280 for (int j = 0; j < BOARD_COLS; j++)
281 {
282 int live_neighbors = 0;
283
284 // Sprawdź sąsiadów komórki
285 for (int x = -1; x <= 1; x++)
286 {
287 for (int y = -1; y <= 1; y++)
288 {
289 if (x == 0 && y == 0)
290 {
291 continue; // Pomijaj aktualną komórkę
292 }
293 int ni = i + x;
294 int nj = j + y;
295 if (ni >= 0 && ni < BOARD_ROWS && nj >= 0 && nj < BOARD_COLS)
296 {
297 live_neighbors += cellsMatrix[ni][nj]; // Zlicz żywych sąsiadów
298 }
299 }
300 }
301
302 // Zasady gry w życie
303 if (cellsMatrix[i][j] == 1) // Jeśli komórka jest żywa
304 {
305 if (live_neighbors < 2 || live_neighbors > 3)
306 {
307 tempMatrix[i][j] = 0; // Umiera z samotności lub przeludnienia
308 }
309 }
310 else // Jeśli komórka jest martwa
311 {
312 if (live_neighbors == 3)
313 {
314 tempMatrix[i][j] = 1; // Ożywa, jeśli ma trzech sąsiadów
315 }
316 }
317 }
318 }
319
320 // Zaktualizuj macierz komórek
321 for (int i = 0; i < BOARD_ROWS; i++)
322 {
323 for (int j = 0; j < BOARD_COLS; j++)
324 {
325 cellsMatrix[i][j] = tempMatrix[i][j];
326 }
327 }
328
329 sfTime waitTime = sfMilliseconds(1000 / simulation->speed); // Oblicz czas oczekiwania na następny krok
330 sfSleep(waitTime); // Poczekaj
331 nextTick(args); // Wywołaj kolejny krok symulacji rekurencyjnie
332}
